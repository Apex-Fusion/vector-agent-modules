# Vector agent swarm — tier-2 schedule (Modules 3 + 6 only).
#
# Module-1 agents (m1-claimer/auditor/juror) are NOT in this file; they run
# under the vector-tier1.service systemd unit because Module-1 protocol
# windows are 30-90 min and cannot be served by 12h cron.
#
# Install with:
#     crontab ~/code/vector-agent-modules/scripts/deploy-agents/crontab.txt
#     crontab -l   # verify
#
# Remove with:
#     crontab -r
#
# Note: cron does NOT expand ${HOME} in command lines on Vixie cron.
# We set HOME= below and use a bare $HOME which Vixie cron *does* pass into
# the spawned shell. run-agent.sh also hard-codes $HOME/vector-agents paths.

SHELL=/bin/bash
HOME=/home/USER_PLACEHOLDER
PATH=/usr/local/bin:/usr/bin:/bin:/home/USER_PLACEHOLDER/.local/bin:/home/USER_PLACEHOLDER/.npm-global/bin

# Minute Hour DayOfMonth Month DayOfWeek  Command
# ── Morning slot — 6 agents, 80 min apart ────────────────────────────────
0  4  * * *  /home/USER_PLACEHOLDER/vector-agents/bin/run-agent.sh m3-staker
20 5  * * *  /home/USER_PLACEHOLDER/vector-agents/bin/run-agent.sh m3-endorser
40 6  * * *  /home/USER_PLACEHOLDER/vector-agents/bin/run-agent.sh m3-challenger
0  8  * * *  /home/USER_PLACEHOLDER/vector-agents/bin/run-agent.sh m6-proposer
20 9  * * *  /home/USER_PLACEHOLDER/vector-agents/bin/run-agent.sh m6-critic
40 10 * * *  /home/USER_PLACEHOLDER/vector-agents/bin/run-agent.sh m6-endorser

# ── Evening slot (same order, +12h) ──────────────────────────────────────
0  16 * * *  /home/USER_PLACEHOLDER/vector-agents/bin/run-agent.sh m3-staker
20 17 * * *  /home/USER_PLACEHOLDER/vector-agents/bin/run-agent.sh m3-endorser
40 18 * * *  /home/USER_PLACEHOLDER/vector-agents/bin/run-agent.sh m3-challenger
0  20 * * *  /home/USER_PLACEHOLDER/vector-agents/bin/run-agent.sh m6-proposer
20 21 * * *  /home/USER_PLACEHOLDER/vector-agents/bin/run-agent.sh m6-critic
40 22 * * *  /home/USER_PLACEHOLDER/vector-agents/bin/run-agent.sh m6-endorser

# IMPORTANT: replace USER_PLACEHOLDER with the actual username running the
# swarm BEFORE installing. bootstrap.sh's 'install-cron' command does this
# substitution automatically and pipes the result into `crontab -`.
